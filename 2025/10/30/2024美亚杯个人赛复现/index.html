import datetime
import time

def timestamp_to_gmt8(timestamp=None):
    """
    将时间戳转换为GMT+8时间
    :param timestamp: 时间戳（秒级），默认为当前时间
    :return: GMT+8时间字符串
    """
    if timestamp is None:
        timestamp = time.time()
    
    # 创建GMT+8时区
    gmt8 = datetime.timezone(datetime.timedelta(hours=8))
    
    # 转换为datetime对象
    if timestamp > 10000000000:  # 毫秒级时间戳
        dt = datetime.datetime.fromtimestamp(timestamp / 1000, tz=gmt8)
    else:  # 秒级时间戳
        dt = datetime.datetime.fromtimestamp(timestamp, tz=gmt8)
    
    return dt.strftime("%Y-%m-%d %H:%M:%S %Z")

# 使用示例
if __name__ == "__main__":
    # 当前时间
    print("当前GMT+8时间:", timestamp_to_gmt8())
    
    # 指定时间戳
    timestamp = 1723625933421  # 2024-01-01 00:00:00 UTC
    print(f"时间戳 {timestamp} 对应的GMT+8时间:", timestamp_to_gmt8(timestamp))
    
    # 毫秒级时间戳
