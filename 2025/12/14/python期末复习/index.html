1.基本数据类型

1.数据类型：

  • 整数(int): 10 , -5 , 0
  • 浮点数(float): 3.14 , -2.5 , 1.0
  • 复数(complex): 3 + 4j
  • 布尔值(bool): True、False

2.序列类型：

  • 字符串(str): “Hello”
  • 列表(list) - 可变序列: [1,2,3,4]
  • 元组(tuple) - 不可变序列: (1,2,3,”a”)

3.集合类型：

  • 集合(set) - 无序、不重复: {1,2,3}

set可变元素不可变。

4.映射类型：

  • 字典 - 键值对: {“name”:”Alex”}

5.特点总结：

类型 可变性 有序性 示例
int 不可变 - 10
float 不可变 - 3.14
str 不可变 有序 “hello”
list 可变 有序 [1,2,3]
tuple 不可变 有序 (1,2,3)
set 可变 无序 {1,2,3}
dict 可变 有序 {“a”:1}

PS:是否具有有序性即能否通过索引访问。

2.for-else结构：

1.基础语法

# for-else 结构
for item in iterable:
    # 循环体代码
    if condition:
        break  # 提前退出循环
else:
    # 循环正常结束时执行（没有break）
    # 这里的 else 不是和 if 配对，而是和 for 配对！

2.简单示例

# 情况1：循环正常结束（执行else）
fruits = ["apple", "banana", "cherry"]

for fruit in fruits:
    if fruit == "grape":
        print("找到葡萄了！")
        break
else:
    print("没有找到葡萄")  # 会执行

# 情况2：循环被break中断（不执行else）
for fruit in fruits:
    if fruit == "banana":
        print("找到香蕉了！")
        break
else:
    print("没有找到香蕉")  # 不会执行
    
# 空列表 - else 会执行
empty_list = []

for item in empty_list:
    print("这不会执行")
    break
else:
    print("循环体根本没执行，但else执行了！")
# 输出：循环体根本没执行，但else执行了！
3.while-else结构：
# 猜数字游戏
import random

secret_number = random.randint(1, 10)
max_attempts = 3
attempt = 0

print("猜数字游戏（1-10），你有3次机会")

while attempt < max_attempts:
    attempt += 1
    guess = int(input(f"第{attempt}次猜: "))
    
    if guess == secret_number:
        print(f"恭喜！猜对了！数字是 {secret_number}")
        break
    elif guess < secret_number:
        print("太小了")
    else:
        print("太大了")
else:
    print(f"游戏结束！数字是 {secret_number}")
4.常见误区：

1.误区：else 总是执行

# 错误理解：以为 else 总是执行
numbers = [1, 2, 3]

for num in numbers:
    print(num)
    # 没有 break
else:
    print("循环结束")  # 这会执行

# 但有 break 时就不执行
for num in numbers:
    if num == 2:
        break
    print(num)
else:
    print("循环结束")  # 不会执行！

2.误区：把else当成 if 的

# 错误：缩进错误
for i in range(3):
    if i == 1:
        print("i等于1")
else:  # 这个 else 属于 for，不是 if！
    print("这会在循环结束后执行")

# 正确：if-else 在循环内
for i in range(3):
    if i == 1:
        print("i等于1")
    else:  # 这个 else 属于 if
        print(f"i不等于1，i={i}")

3.continue不影响else

# continue 不会阻止 else 执行
for i in range(5):
    if i % 2 == 0:
        continue  # 跳过偶数
    print(f"奇数: {i}")
else:
    print("循环结束，else 执行了！")  # 会执行
# continue 不是 break，循环正常结束
4.总结：
特性 说明
执行时机 循环正常结束(没有break)时执行
空循环 也会执行else
continue 不影响else执行
嵌套循环 只受当前循环的break影响
主要用途 查找元素、验证条件、清理工作

2.文件打开模式

# 基本模式
"""
r:  只读（默认），文件必须存在
w:  只写，会覆盖原文件，文件不存在则创建
a:  追加，文件不存在则创建
x:  独占创建，文件必须不存在
b:  二进制模式，如 'rb', 'wb'
t:  文本模式（默认），如 'rt', 'wt'
+： 读写模式，如 'r+', 'w+', 'a+'
"""

3.打开模式示例

with open('test.txt', 'w') as f:

    f.write("第二行\n")

# 追加内容
with open('test.txt', 'a') as f:
    f.write("第三行（追加）\n")

# 读写模式
with open('test.txt', 'r+') as f:
    content = f.read()
    f.write("新内容\n")

# 二进制模式（处理图片、音频等）
with open('image.jpg', 'rb') as f:
    image_data = f.read()

# 文本模式指定编码
with open('file.txt', 'r', encoding='utf-8') as f:
    content = f.read()

    print(f"第一次读取10字节: {chunk1}")  # 0123456789
    
    # 再读取20个字符
    chunk2 = f.read(20)
    print(f"第二次读取20字节: {chunk2}")  # 01234567890123456789
    
    # 继续读取，不指定大小则读取剩余所有
    rest = f.read()
    print(f"剩余长度: {len(rest)}")  # 70个字符

# 读取大文件的推荐方式
def read_large_file(filename, chunk_size=1024):
    """分块读取大文件"""
    with open(filename, 'r') as f:
        while True:
            chunk = f.read(chunk_size)
            if not chunk:  # 读取到文件末尾
                break
            # 处理每一块数据
            yield chunk

# 使用示例
for chunk in read_large_file('large_file.txt', 1024):
    process_chunk(chunk)

2.readline() - 逐行读取

# 创建多行文件
with open('multiline.txt', 'w') as f:
    for i in range(5):
        f.write(f"这是第{i+1}行\n")

# 逐行读取
with open('multiline.txt', 'r') as f:
    line1 = f.readline()
    print(f"第1行: {line1.strip()}")  # 这是第1行
    
    line2 = f.readline()
    print(f"第2行: {line2.strip()}")  # 这是第2行
    
    # 继续读取剩余行
    while True:
        line = f.readline()
        if not line:  # 空字符串表示文件结束
            break
        print(f"继续读取: {line.strip()}")

# 使用 readline 读取整个文件
def read_file_line_by_line(filename):
    """逐行读取整个文件"""
    with open(filename, 'r') as f:
        line_number = 1
        while True:
            line = f.readline()
            if not line:
                break
            print(f"行 {line_number}: {line.strip()}")
            line_number += 1

with open('seek_test.txt', 'r') as f:
    # 从开头读取5个字符
    print(f"初始位置: {f.tell()}")  # 0
    data1 = f.read(5)
    print(f"读取5字节后: {data1}")  # 01234
    print(f"当前位置: {f.tell()}")   # 5
    
    # 移动到开头
    f.seek(0)
    print(f"seek(0)后位置: {f.tell()}")  # 0
    
    # 移动到第10个字节
    f.seek(10)
    print(f"seek(10)后读取: {f.read(5)}")  # 5678a
    
    # 相对当前位置移动
    f.seek(5, 1)  # 从当前位置向后移动5
    print(f"从当前位置+5后: {f.read(5)}")  # cdefg
    
    # 从文件末尾移动（通常用于二进制文件）
    f.seek(-10, 2)  # 从末尾向前10个字节

2.类方法

class Dog:
    species = "Canis familiaris"
    total_dogs = 0  # 记录创建的狗的数量
    
    def __init__(self, name, age):
        self.name = name
        self.age = age
        Dog.total_dogs += 1
    
    # 类方法：第一个参数是cls，用@classmethod装饰
    @classmethod
    def get_total_dogs(cls):
        return cls.total_dogs
    
    @classmethod
    def change_species(cls, new_species):
        cls.species = new_species
    
    @classmethod
    def create_from_birth_year(cls, name, birth_year):
        """替代构造函数，根据出生年份创建"""
        from datetime import datetime
        current_year = datetime.now().year
        age = current_year - birth_year
        return cls(name, age)

# 使用类方法
print(Dog.get_total_dogs())  # 0

dog1 = Dog("Buddy", 3)
dog2 = Dog("Max", 5)

print(Dog.get_total_dogs())  # 2

# 修改类属性
Dog.change_species("Canis lupus")
print(dog1.species)  # Canis lupus

# 使用类方法作为替代构造函数
dog3 = Dog.create_from_birth_year("Charlie", 2020)
print(f"{dog3.name} is {dog3.age} years old")

3.静态方法

class Calculator:
    @staticmethod
    def add(x, y):
        """静态方法：与类相关但不依赖类或实例"""
        return x + y
    
    @staticmethod
    def multiply(x, y):
        return x * y

# 使用静态方法（无需创建实例）
print(Calculator.add(10, 5))      # 15
print(Calculator.multiply(10, 5)) # 50

# 也可以用实例调用
calc = Calculator()
print(calc.add(3, 4))  # 7

3.带权重的随机选择

# 权重随机选择
fruits = ['apple', 'banana', 'orange']
weights = [10, 1, 5]  # apple:10, banana:1, orange:5

# 按权重随机选择
choices = random.choices(fruits, weights=weights, k=10)
print("10次权重选择:", choices)

# 计算频率
from collections import Counter
freq = Counter(choices)
print("选择频率:", freq)

# 累积权重（更高效）
cum_weights = [10, 11, 16]  # 累积权重
choices2 = random.choices(fruits, cum_weights=cum_weights, k=1000)
freq2 = Counter(choices2)
print("\n1000次选择频率:", freq2)

函数 描述 示例 返回值
random.seed(a=None) 初始化随机数生成器 random.seed(42) None
random.random() 生成 [0.0, 1.0) 之间的随机浮点数 random.random() 0.374540…
random.uniform(a, b) 生成 [a, b] 之间的随机浮点数 random.uniform(1, 10) 5.678
random.randint(a, b) 生成 [a, b] 之间的随机整数 random.randint(1, 100) 42

random.choices(population, k=1) 从序列中随机选择k个元素（可重复） random.choices([1,2,3], k=2) [3, 1]
random.sample(population, k) 从序列中随机选择k个元素（不重复） random.sample([1,2,3,4], 2) [3, 4]
random.shuffle(seq) 将序列顺序随机打乱（原地修改） random.shuffle([1,2,3]) [2, 3, 1]

7.time库

1.示例

# 时间戳转字符串
timestamp = time.time()
print(f"时间戳: {timestamp}")

# 转可读字符串
readable = time.ctime(timestamp)
print(f"可读格式: {readable}")

# 转格式化字符串
local = time.localtime(timestamp)
formatted = time.strftime("%Y-%m-%d %H:%M:%S", local)
print(f"格式化: {formatted}")

# 字符串转时间戳
date_str = "2024-01-15 14:30:00"
time_tuple = time.strptime(date_str, "%Y-%m-%d %H:%M:%S")
timestamp_from_str = time.mktime(time_tuple)
print(f"字符串转时间戳: {date_str} -> {timestamp_from_str}")

# 验证转换
print(f"验证: {time.ctime(timestamp_from_str)}")

2.总结

time() 当前时间戳 time.time()

strftime() 时间格式化 time.strftime(“%Y-%m-%d”)
strptime() 时间解析 time.strptime(“2024-01-01”, “%Y-%m-%d”)
ctime() 可读时间 time.ctime()
perf_counter() 高精度计时 time.perf_counter()

9.lambda函数

1.基本语法和定义

# 基本语法：
# lambda 参数: 表达式

# 普通函数 vs lambda函数
# 1. 普通函数
def add_regular(x, y):
    return x + y

# 2. lambda函数
add_lambda = lambda x, y: x + y

print(f"普通函数: add_regular(3, 5) = {add_regular(3, 5)}")   # 8
print(f"lambda函数: add_lambda(3, 5) = {add_lambda(3, 5)}")   # 8

# 匿名使用（不赋值给变量）
result = (lambda x, y: x * y)(4, 5)
print(f"直接调用: (lambda x, y: x*y)(4,5) = {result}")  # 20

2.示例

# map() - 映射
numbers = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x ** 2, numbers))
print(f"map平方: {squared}")  # [1, 4, 9, 16, 25]

# filter() - 过滤
even_numbers = list(filter(lambda x: x % 2 == 0, numbers))
print(f"filter偶数: {even_numbers}")  # [2, 4]

# reduce() - 累积（需要导入）
from functools import reduce
product = reduce(lambda x, y: x * y, numbers)
print(f"reduce乘积: {product}")  # 120

# sorted() - 排序
students = [
    {"name": "Alice", "score": 85},
    {"name": "Bob", "score": 92},
    {"name": "Charlie", "score": 78}
]

# 按分数排序
sorted_by_score = sorted(students, key=lambda s: s["score"])
print("按分数排序:")
for student in sorted_by_score:
    print(f"  {student['name']}: {student['score']}")

# 按名字长度排序
sorted_by_name_length = sorted(students, key=lambda s: len(s["name"]))
print("按名字长度排序:")
for student in sorted_by_name_length:
    print(f"  {student['name']}: {len(student['name'])}个字符")
