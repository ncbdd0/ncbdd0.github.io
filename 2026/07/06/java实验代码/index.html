3、类的静态多态性（即方法重载）应用

public class MathCalculator {


    public int funSum(int a, int b) {


        return a + b;


    }




    public int funSum(int a, int b, int c) {


        return a + b + c;


    }




    public int funMul(int a, int b) {


        return a * b;


    }




    public int funMul(int a, int b, int c) {


        return a * b * c;


    }




    public static void main(String[] args) {


        MathCalculator calc = new MathCalculator();


        System.out.println("funSum(2, 3) = " + calc.funSum(2, 3)); // 5


        System.out.println("funSum(1, 2, 3) = " + calc.funSum(1, 2, 3)); // 6


        System.out.println("funMul(4, 5) = " + calc.funMul(4, 5)); // 20


        System.out.println("funMul(2, 3, 4) = " + calc.funMul(2, 3, 4)); // 24


    }


}

4.数组的应用

将两个给定的成绩数组合并为一个新数组，并对新数组进行排序、求最高成绩和平均成绩等操作。

import java.util.Arrays;




public class Grade {


    public static void main(String[] args) throws Exception {


        int classOne[] = new int[] { 70, 79, 90, 71, 87, 73, 77, 86, 83, 76 };


        int classTwo[] = new int[] { 99, 89, 78, 85, 80, 91, 97, 74, 75, 88, 92, 94 };


        int grade[] = new int[22];


        int index = 0;


        int i;


        // 合并数组


        for (i = 0; i < 10; i++) {


            grade[index] = classOne[i];


            index++;


        }


        for (i = 0; i < 12; i++) {


            grade[index] = classTwo[i];


            index++;


        }


        // 排序


        Arrays.sort(grade);


        System.out.println("排序后的数组为：" + Arrays.toString(grade));


        // 输出最高成绩和平均成绩


        int max_num = grade[grade.length - 1], sum = 0;


        for (i = 0; i < 22; i++) {


            sum += grade[i];


        }


        int average = sum / grade.length;


        System.out.println("最高成绩为：" + max_num + "，平均成绩为：" + average);


    }


}

输出：

排序后的数组为：[70, 71, 73, 74, 75, 76, 77, 78, 79, 80, 83, 85, 86, 87, 88, 89, 90, 91, 92, 94, 97, 99]
最高成绩为：99，平均成绩为：83

6、泛型的应用

修改给定代码，使得数组排序结果为从大到小进行排序，并打印输出排序结果。

import java.util.Arrays;




public class SortArray {


    public static <T extends Comparable<T>> void sort(T[] arr) {


        boolean flag = false;


        for (int i = 0; i < arr.length - 1; i++) {


            for (int j = 1; j < arr.length - i; j++) {


                if (arr[j].compareTo(arr[j - 1]) < 0) {


                    flag = true;


                    T temp = arr[j];


                    arr[j] = arr[j - 1];


                    arr[j - 1] = temp;


                }


            }


            if (flag == false) {


                return;


            } else {


                flag = false;


            }


        }


    }




    public static void main(String[] args) {


        Integer[] arr1 = { 1, 2, 8, 6, 14, 3, 64, 13 };


        String[] arr2 = { "a", "u", "c", "x", "v" };


        SortArray.sort(arr1);


        SortArray.sort(arr2);


        System.out.println(Arrays.toString(arr1));


        System.out.println(Arrays.toString(arr2));


    }


}

输出：

[1, 2, 3, 6, 8, 13, 14, 64]
[a, c, u, v, x]

7、集合类ArrayList的应用

实现集合的创建，并进行元素的添加、遍历、删除等操作。

import java.util.ArrayList;




public class TestArrayList {


    public static void main(String[] args) {


        ArrayList<String> strNames = new ArrayList<String>();


        strNames.add("橘子");


        strNames.add("荔枝");


        strNames.add("草莓");


        strNames.add("芒果");


        strNames.add("苹果");


        // 第一种遍历


        for (String name : strNames) {


            System.out.print(name + " ");


        }


        System.out.println("以上是第一种遍历方法");


        // 第二种遍历


        String newnames[] = new String[strNames.size()];


        strNames.toArray(newnames);


        for (int i = 0; i < newnames.length; i++) {


            System.out.print(newnames[i] + " ");


        }


        System.out.println("以上是第二种遍历方法");


        // 第三种遍历


        for (int i = 0; i < strNames.size(); i++) {


            System.out.print(strNames.get(i) + " ");


        }


        System.out.println("以上是第三种遍历方法");


        String delnum = strNames.remove(2); // 删除第三个元素


        System.out.println("删除的元素是：" + delnum);


        System.out.println("目前的第三个元素是：" + strNames.get(2));


        System.out.println("集合的大小为：" + strNames.size());




    }


}

输出：

橘子 荔枝 草莓 芒果 苹果 以上是第一种遍历方法
橘子 荔枝 草莓 芒果 苹果 以上是第二种遍历方法
橘子 荔枝 草莓 芒果 苹果 以上是第三种遍历方法

8.课程成绩的异常处理

给定或从键盘输入某个Java班10位同学的课程成绩，分别统计及格人数、不及格人数和平均分并打印输出。设计一个异常类，在统计时当课程成绩中有人成绩小于0或大于100分时则抛出异常，并作出相应处理。

import java.util.Scanner;




class gradeexc extends Exception {


    public gradeexc() {


        System.out.print("成绩异常，");


    }


}




public class Test {


    public static void main(String[] args) throws gradeexc {


        Scanner input = new Scanner(System.in);


        int[] grade = new int[10];


        for (int i = 0; i < 10; i++) {


            System.out.print("请输入第" + (i + 1) + "位同学的成绩");


            grade[i] = Integer.parseInt(input.nextLine());


        }


        int out_60 = 0, less_60 = 0, sum = 0;


        for (int j = 0; j < 10; j++) { // 统计成绩


            try {


                if (grade[j] < 0 || grade[j] > 100)


                    throw new gradeexc();


            } catch (gradeexc e) {


                System.out.println("请重新输入第" + (j + 1) + "位同学的成绩");


                grade[j] = Integer.parseInt(input.nextLine());


            } finally {


                if (grade[j] > 0 && grade[j] < 100) {


                    sum += grade[j];


                    if (grade[j] >= 60) {


                        out_60 += 1;


                    } else {


                        less_60 += 1;


                    }


                } else {


                    j -= 1;


                }


            }


        }


        int avg = sum / 10;


        System.out.println("及格同学的人数为:" + out_60 + "，不及格同学的人数为:" + less_60 + "，平均分为:" + avg);


    }


}

10.带缓存输入流的应用

import java.io.BufferedInputStream;


import java.io.FileInputStream;


import java.util.Date;




public class J_BufferedInputStream {


    private static String m_fileName = "bin/J_BufferedInputStream.class";




    public static void main(String[] args) {


        try {


            int i, ch;


            i = 0;


            Date d1 = new Date();


            FileInputStream f = new FileInputStream(m_fileName);


            while ((ch = f.read()) != -1)


                i++;


            f.close();


            Date d2 = new Date();


            long t = d2.getTime() - d1.getTime();


            System.out.printf("读取文件%1$s(共%2$d字节)%n",


                    m_fileName, i);


            System.out.printf("不带缓存的方法需要%1$d毫秒%n", t);


            i = 0;


            d1 = new Date();


            f = new FileInputStream(m_fileName);


            BufferedInputStream fb = new BufferedInputStream(f);


            while ((ch = fb.read()) != -1)


                i++;


            fb.close();


            d2 = new Date();


            t = d2.getTime() - d1.getTime();


            System.out.printf("带缓存的方法需要%1$d毫秒%n", t);


        } catch (Exception e) {


            System.err.println("发生异常:" + e);


            e.printStackTrace();


        }


    }


}
